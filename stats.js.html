/**
 * Closes the statistics pane.

function closeStatsPane () {
  timerBlock.classList.remove('slide-open');
  statsPane.classList.remove('slide-open');
  timerBlock.classList.add('slide-close');
  statsPane.classList.add('slide-close');
}
